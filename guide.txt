Изменить manimlib\utils\tex_file_writing.py

# Write tex file
tex_path.write_text(full_tex, encoding="utf-8")

# Run latex compiler
process = subprocess.run(
    [
        compiler,
        # "-no-pdf",
        "-interaction=batchmode",
        "-halt-on-error",
        f"-output-directory={temp_dir}",
        tex_path
    ],
    capture_output=True,
    text=True
)
